;Display'DBCET' in the Row 1 and Column 5 and 'ADBU' in the Row 2 and Column 6 of 16x2 LCD and then blink DBCET and ADBU alternately.
 
HERE:MOV A, #38H; To intialize 16x2 LCD
ACALL COMNWRT
ACALL DELAY
MOV A, #0CH; DISPLAY ON
ACALL COMNWRT
ACALL DELAY
MOV A, #01H; Clear display screen
ACALL COMNWRT
ACALL DELAY
MOV A, #06H; Increment Cursor
ACALL COMNWRT
ACALL DELAY
ACALL REPEAT
ACALL DELAY1
ACALL REPEAT1
ACALL DELAY1
SJMP HERE

REPEAT:MOV A, #84H
ACALL COMNWRT
ACALL DELAY
MOV A, #'D'
ACALL DATAWRT
ACALL DELAY
MOV A, #'B'
ACALL DATAWRT
ACALL DELAY
MOV A, #'C'
ACALL DATAWRT
ACALL DELAY
MOV A, #'E'
ACALL DATAWRT
ACALL DELAY
MOV A, #'T'
ACALL DATAWRT
ACALL DELAY1
MOV A, #01H; Display Off
ACALL COMNWRT
ACALL DELAY
RET

REPEAT1:MOV A, #0C5H
ACALL COMNWRT
ACALL DELAY
MOV A, #'A'
ACALL DATAWRT
ACALL DELAY
MOV A, #'D'
ACALL DATAWRT
ACALL DELAY
MOV A, #'B'
ACALL DATAWRT
ACALL DELAY
MOV A, #'U'
ACALL DATAWRT
ACALL DELAY
RET
COMNWRT: MOV P1,A
CLR P3.5
SETB P3.7
ACALL DELAY
CLR P3.7
RET
DATAWRT:MOV P1,A
SETB P3.5
SETB P3.7
ACALL DELAY
CLR P3.7
RET 
DELAY: MOV R1,#0FFH
L1:DJNZ R1,L1
RET
DELAY1:MOV R2,#0FFH
L3:MOV R3,#0FFH
L2:DJNZ R3,L2
DJNZ R2,L3
RET
